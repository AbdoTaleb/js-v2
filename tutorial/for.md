for
==============================



```
/**
 * Count to nine from zero 
 **/
for(var i=0; i < 10; i++) {
	console.log(i);
}

// Same as abone but to ten
for(var i=0; i <= 10; i++) {
	console.log(i);
}

// Not including zero

for(var i=0; i <= 10; ++i) {
	console.log(i);
}


```
