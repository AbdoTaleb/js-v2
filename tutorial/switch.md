switch
==============================



```
/**
 * Find the price of item 2
 */
var x = 2;

switch(x) {
	case 0:
		console.log('Item 0: 3$')
	case 1:
		console.log('Item 1: 6$')
	case 2:
		console.log('Item 2: 9$')
	case 3:
		console.log('Item 3: 5$')
}

```
